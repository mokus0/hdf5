cmake_minimum_required (VERSION 2.8.10)
PROJECT (HDF5_HL_CPP_TEST)

# Setup include Directories
#-----------------------------------------------------------------------------

    ${HDF5_TEST_LIB_TARGET}
    ${HDF5_CPP_LIB_TARGET}
    ${HDF5_HL_LIB_TARGET}
    ${HDF5_HL_CPP_LIB_TARGET}
)
